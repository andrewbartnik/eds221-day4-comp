calc_age <- function(animal, human_age) {
  if(animal == "dog") {
    print(human_age * 7)
  } else if(animal == "goat") {
    print(human_age * 4.7)
  }
  else print(paste0("This function is not built to calculate ages of ", animal))
}

#shortcut - cmd/opt/x

logistic_growth <- function(K, N0, r, time) {
  K / (1 + ((K - N0) / N0) * exp(-r * time))
}

  0.3 * rho * pi * (radius ^ 2) * (windspeed ^ 3)

}
